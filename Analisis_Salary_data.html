Creamos los rangos de edades, calculamos su promedio por edades y cuántas personas hay en cada rango.

In [34]:
edad_rangos = pd.cut(df['Age'], 
                     bins=[20, 30, 40, 50, 65],
                     labels=['21-30 años', '31-40 años', '41-50 años', '51-62 años'])
salario_por_edad = df.groupby(edad_rangos, observed=True)['Salary'].mean().sort_index()

print("\nSalario promedio por rango de edad:")
print(salario_por_edad)

personas_por_edad = df.groupby(edad_rangos, observed=True).size()
print("\nCantidad de personas por rango:")
print(personas_por_edad)
